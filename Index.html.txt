<script>
  const ROMAN = ['I','II','III','IV','V','VI','VII','VIII','IX','X','XI','XII','XIII','XIV','XV','XVI','XVII','XVIII','XIX','XX','XXI','XXII','XXIII','XXIV','XXV'];
  const TARGET = 200;
  const STORAGE_KEY = 'flip7_vengeance_v5';

  const NUMBER_CARDS = [
    { val: 0, qty: 1 }, { val: 1, qty: 1 }, { val: 2, qty: 2 }, { val: 3, qty: 3 },
    { val: 4, qty: 4 }, { val: 5, qty: 5 }, { val: 6, qty: 6 },
    { val: 7, qty: 7, unlucky: true }, { val: 8, qty: 8 }, { val: 9, qty: 9 },
    { val: 10, qty: 10 }, { val: 11, qty: 11 }, { val: 12, qty: 12 },
    { val: 13, qty: 13, lucky: true }
  ];

  const MODIFIERS = [
    { id: '+2', val: 2, type: 'add', label: '+2' },
    { id: '+4', val: 4, type: 'add', label: '+4' },
    { id: '+6', val: 6, type: 'add', label: '+6' },
    { id: '+8', val: 8, type: 'add', label: '+8' },
    { id: '+10', val: 10, type: 'add', label: '+10' },
    { id: 'x2', val: 2, type: 'mult', label: 'x2' },
    { id: '/2', val: 2, type: 'div', label: '\u00f72' },
    { id: '-2', val: -2, type: 'add', label: '-2', neg: true },
    { id: '-4', val: -4, type: 'add', label: '-4', neg: true },
    { id: '-6', val: -6, type: 'add', label: '-6', neg: true },
    { id: '-8', val: -8, type: 'add', label: '-8', neg: true },
    { id: '-10', val: -10, type: 'add', label: '-10', neg: true }
  ];

  let state = {};

  function lsSave() {
    try { localStorage.setItem(STORAGE_KEY, JSON.stringify(state)); } catch(e) {}
  }
  function lsLoad() {
    try {
      const raw = localStorage.getItem(STORAGE_KEY);
      if (raw) { state = JSON.parse(raw); return true; }
    } catch(e) {}
    return false;
  }

  function defaultState() {
    state = {
      players: [
        { id: 1, name: 'Player 1', total: 0, history: [], current: emptyCurrent() },
        { id: 2, name: 'Player 2', total: 0, history: [], current: emptyCurrent() },
        { id: 3, name: 'Player 3', total: 0, history: [], current: emptyCurrent() },
        { id: 4, name: 'Player 4', total: 0, history: [], current: emptyCurrent() }
      ],
      activeId: 1, round: 1, gameOver: false, undoStack: []
    };
  }
  function emptyCurrent() {
    return { numbers: [], modifiers: [], busted: false, flip7: false, locked: false };
  }
  function getActive() {
    return state.players.find(p => p.id === state.activeId) || state.players[0];
  }
  function numberCount(arr, val) {
    return arr.filter(n => n === val).length;
  }

  function computeScore(p) {
    const c = p.current;
    if (c.busted) return { total: 0, breakdown: [{ label: 'Busted', val: '0' }] };
    const breakdown = [];
    const numSum = c.numbers.reduce((s, n) => s + n, 0);
    if (c.numbers.length) {
      breakdown.push({ label: 'Numbers', val: c.numbers.join(' + ') + ' = ' + numSum });
    }
    let working = numSum;
    if (c.modifiers.includes('/2')) {
      working = Math.floor(working / 2);
      breakdown.push({ label: 'Halve', val: '\u2192 ' + working });
    }
    if (c.modifiers.includes('x2')) {
      working = working * 2;
      breakdown.push({ label: 'Double', val: '\u2192 ' + working });
    }
    const adds = c.modifiers.filter(m => m !== '/2' && m !== 'x2');
    if (adds.length) {
      const addSum = adds.reduce((s, a) => {
        const m = MODIFIERS.find(x => x.id === a);
        return s + (m ? m.val : 0);
      }, 0);
      working += addSum;
      breakdown.push({ label: 'Modifiers', val: adds.join(' ') + ' = ' + (addSum >= 0 ? '+' : '') + addSum });
    }
    if (c.flip7) {
      working += 15;
      breakdown.push({ label: 'Flip 7 bonus', val: '+15' });
    }
    if (working < 0) working = 0;
    return { total: working, breakdown };
  }

  function render() {
    document.getElementById('round-num').textContent = ROMAN[state.round - 1] || String(state.round);
    renderStandings();
    renderActivePanel();
    renderNumberGrid();
    renderModifierGrid();
    renderBanner();
  }

  function renderStandings() {
    const el = document.getElementById('standings');
    el.innerHTML = '';
    const maxTotal = Math.max(...state.players.map(p => p.total), 0);
    const leaderId = (maxTotal > 0)
      ? state.players.find(p => p.total === maxTotal).id
      : null;

    state.players.forEach(p => {
      const div = document.createElement('div');
      div.className = 'standing'
        + (p.id === state.activeId ? ' active' : '')
        + (p.id === leaderId ? ' leading' : '');
      const c = p.current;
      let pendingText = '';
      let pendingClass = '';
      if (c.busted) { pendingText = 'BUST'; pendingClass = 'bust'; }
      else if (c.flip7) { pendingText = 'FLIP 7'; pendingClass = 'flip7'; }
      else if (c.numbers.length || c.modifiers.length) {
        const s = computeScore(p);
        pendingText = '+' + s.total;
      }
      div.innerHTML =
        '<div class="standing-name">' + escapeHtml(p.name) + '</div>' +
        '<div class="standing-score">' + p.total + '</div>' +
        '<div class="standing-pending ' + pendingClass + '">' + pendingText + '</div>';
      div.addEventListener('click', () => {
        state.activeId = p.id;
        lsSave();
        render();
      });
      el.appendChild(div);
    });
  }

  function renderActivePanel() {
    const p = getActive();
    if (!p) return;
    document.getElementById('active-name').value = p.name;

    const lineEl = document.getElementById('card-line');
    lineEl.innerHTML = '';

    p.current.numbers.forEach((n, idx) => {
      const card = document.createElement('div');
      card.className = 'placed-card';
      const count = numberCount(p.current.numbers, n);
      const allowed = (n === 13) ? 2 : 1;
      if (count > allowed) card.classList.add('duplicate');
      card.textContent = n;
      card.addEventListener('click', () => {
        p.current.numbers.splice(idx, 1);
        const unique = new Set(p.current.numbers).size;
        if (unique < 7) p.current.flip7 = false;
        let stillBusted = false;
        const counts = {};
        p.current.numbers.forEach(x => counts[x] = (counts[x] || 0) + 1);
        for (const k in counts) {
          const allowedHere = (parseInt(k,10) === 13) ? 2 : 1;
          if (counts[k] > allowedHere) stillBusted = true;
        }
        if (!stillBusted) p.current.busted = false;
        lsSave();
        render();
      });
      lineEl.appendChild(card);
    });

    p.current.modifiers.forEach((mid, idx) => {
      const m = MODIFIERS.find(x => x.id === mid);
      if (!m) return;
      const card = document.createElement('div');
      card.className = 'placed-card modifier';
      if (m.neg) card.classList.add('negmod');
      if (mid === 'x2' || mid === '/2') card.classList.add('special');
      card.textContent = m.label;
      card.addEventListener('click', () => {
        p.current.modifiers.splice(idx, 1);
        lsSave();
        render();
      });
      lineEl.appendChild(card);
    });

    document.getElementById('flag-bust').classList.toggle('active', p.current.busted);
    document.getElementById('flag-flip7').classList.toggle('active', p.current.flip7);
    document.getElementById('flag-locked').classList.toggle('active', p.current.locked);

    const scoreObj = computeScore(p);
    const scoreEl = document.getElementById('round-score');
    scoreEl.classList.toggle('bust', p.current.busted);
    scoreEl.classList.toggle('flip7', p.current.flip7);
    document.getElementById('round-score-num').textContent = scoreObj.total;

    const bdEl = document.getElementById('breakdown');
    if (p.current.busted) {
      bdEl.innerHTML = '<div class="row total bust"><span>Busted!</span><span>0</span></div>';
    } else if (!p.current.numbers.length && !p.current.modifiers.length) {
      bdEl.innerHTML = '<div class="row" style="opacity:0.6"><span>Awaiting cards...</span><span>0</span></div>';
    } else {
      let html = '';
      scoreObj.breakdown.forEach(b => {
        html += '<div class="row"><span>' + b.label + '</span><span>' + b.val + '</span></div>';
      });
      html += '<div class="row total"><span>Total</span><span>' + scoreObj.total + '</span></div>';
      bdEl.innerHTML = html;
    }
  }

  function renderNumberGrid() {
    const p = getActive();
    const grid = document.getElementById('number-grid');
    grid.innerHTML = '';
    NUMBER_CARDS.forEach(nc => {
      const used = numberCount(p.current.numbers, nc.val);
      const div = document.createElement('button');
      div.className = 'num-card n' + nc.val;
      const disabled = used >= nc.qty || p.current.busted;
      if (disabled) div.classList.add('disabled');

      div.innerHTML =
        '<div class="corner tl">' + nc.val + '</div>' +
        '<div class="num-val">' + nc.val + '</div>' +
        '<div class="corner br">' + nc.val + '</div>' +
        '<div class="count">' + used + '/' + nc.qty + '</div>';

      div.addEventListener('click', () => {
        if (disabled) return;
        p.current.numbers.push(nc.val);
        const unique = new Set(p.current.numbers).size;
        if (unique >= 7 && !p.current.flip7) p.current.flip7 = true;
        const totalSame = numberCount(p.current.numbers, nc.val);
        const allowed = nc.lucky ? 2 : 1;
        if (totalSame > allowed && !p.current.busted) {
          p.current.busted = true;
          p.current.flip7 = false;
        }
        lsSave();
        render();
      });
      grid.appendChild(div);
    });
  }

  function renderModifierGrid() {
    const p = getActive();
    const grid = document.getElementById('modifier-grid');
    grid.innerHTML = '';
    MODIFIERS.forEach(m => {
      const used = p.current.modifiers.includes(m.id);
      const btn = document.createElement('button');
      btn.className = 'mod-card';
      if (m.neg) btn.classList.add('neg');
      if (m.id === '/2') btn.classList.add('div');
      if (m.id === 'x2') btn.classList.add('x2');
      if (used || p.current.busted) btn.classList.add('disabled');
      const labelText = m.type === 'mult' ? 'Double' : m.type === 'div' ? 'Halve' : (m.neg ? 'Penalty' : 'Bonus');
      btn.innerHTML = '<div class="mod-val">' + m.label + '</div><span class="label">' + labelText + '</span>';
      btn.addEventListener('click', () => {
        if (used || p.current.busted) return;
        p.current.modifiers.push(m.id);
        lsSave();
        render();
      });
      grid.appendChild(btn);
    });
  }

  function renderBanner() {
    const banner = document.getElementById('banner');
    if (state.gameOver) {
      const winner = [...state.players].sort((a,b) => b.total - a.total)[0];
      banner.classList.add('show');
      document.getElementById('winner-name').textContent = winner.name;
      document.getElementById('winner-score').textContent = winner.total + ' pts \u00b7 ' + (state.round - 1) + ' rounds';
      document.getElementById('commit-btn').disabled = true;
    } else {
      banner.classList.remove('show');
      document.getElementById('commit-btn').disabled = false;
    }
  }

  function escapeHtml(s) {
    return String(s).replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]));
  }

  document.getElementById('active-name').addEventListener('change', e => {
    const p = getActive();
    if (p) { p.name = e.target.value.trim() || 'Player'; lsSave(); render(); }
  });
  document.getElementById('active-name').addEventListener('focus', e => e.target.select());

  document.querySelectorAll('.status-flag').forEach(el => {
    el.addEventListener('click', () => {
      const p = getActive();
      if (!p) return;
      const flag = el.dataset.flag;
      if (flag === 'bust') {
        p.current.busted = !p.current.busted;
        if (p.current.busted) p.current.flip7 = false;
      } else if (flag === 'flip7') {
        p.current.flip7 = !p.current.flip7;
        if (p.current.flip7) p.current.busted = false;
      } else if (flag === 'locked') {
        p.current.locked = !p.current.locked;
      }
      lsSave();
      render();
    });
  });

  function twoTapConfirm(btn, armedLabel, onConfirm) {
    if (btn._armed) {
      clearTimeout(btn._disarm);
      btn._armed = false;
      onConfirm();
      return;
    }
    btn._armed = true;
    btn._origLabel = btn.innerHTML;
    btn.innerHTML = armedLabel;
    btn._origBg = btn.style.background;
    btn._origColor = btn.style.color;
    btn.style.background = 'var(--c-red)';
    btn.style.color = 'var(--paper)';
    btn._disarm = setTimeout(() => {
      btn._armed = false;
      btn.innerHTML = btn._origLabel;
      btn.style.background = btn._origBg;
      btn.style.color = btn._origColor;
    }, 2500);
  }

  document.getElementById('clear-btn').addEventListener('click', e => {
    const p = getActive();
    if (!p) return;
    if (!(p.current.numbers.length || p.current.modifiers.length || p.current.busted || p.current.flip7)) return;
    twoTapConfirm(e.currentTarget, 'Confirm?', () => {
      p.current = emptyCurrent();
      lsSave();
      render();
    });
  });

  function addNewPlayer() {
    const newId = Math.max(0, ...state.players.map(p => p.id)) + 1;
    state.players.push({
      id: newId, name: 'Player ' + newId,
      total: 0, history: [], current: emptyCurrent()
    });
    state.activeId = newId;
    lsSave();
    render();
    return newId;
  }

  document.getElementById('quick-add-btn').addEventListener('click', addNewPlayer);

  const rosterModal = document.getElementById('roster-modal');
  const rosterList = document.getElementById('roster-list');

  function openRosterModal() {
    renderRosterList();
    rosterModal.classList.add('show');
    document.body.style.overflow = 'hidden';
    document.querySelector('.action-bar').style.display = 'none';
  }
  function closeRosterModal() {
    rosterModal.classList.remove('show');
    document.body.style.overflow = '';
    document.querySelector('.action-bar').style.display = '';
    render();
  }

  function renderRosterList() {
    document.getElementById('player-count').textContent = state.players.length;
    rosterList.innerHTML = '';
    state.players.forEach((p, idx) => {
      const row = document.createElement('div');
      row.className = 'roster-row';
      const canRemove = state.players.length > 2;
      row.innerHTML =
        '<span class="badge">' + (idx + 1) + '</span>' +
        '<input type="text" class="name-input" value="' + escapeHtml(p.name) + '" maxlength="18" data-id="' + p.id + '" />' +
        '<span class="total-pill">' + p.total + '</span>' +
        '<button class="remove-x" data-id="' + p.id + '" ' + (canRemove ? '' : 'disabled') + ' title="Remove">&#10005;</button>';
      rosterList.appendChild(row);
    });

    rosterList.querySelectorAll('.name-input').forEach(el => {
      el.addEventListener('change', e => {
        const id = +e.target.dataset.id;
        const p = state.players.find(x => x.id === id);
        if (p) {
          p.name = e.target.value.trim() || 'Player';
          lsSave();
          if (state.activeId === id) {
            document.getElementById('active-name').value = p.name;
          }
        }
      });
      el.addEventListener('focus', e => e.target.select());
    });

    rosterList.querySelectorAll('.remove-x').forEach(el => {
      el.addEventListener('click', e => {
        if (state.players.length <= 2) return;
        const btn = e.currentTarget;
        const id = +btn.dataset.id;

        if (!btn.classList.contains('confirm')) {
          rosterList.querySelectorAll('.remove-x.confirm').forEach(b => {
            b.classList.remove('confirm');
            b.innerHTML = '&#10005;';
          });
          btn.classList.add('confirm');
          btn.textContent = 'Confirm?';
          btn._disarmTimer = setTimeout(() => {
            btn.classList.remove('confirm');
            btn.innerHTML = '&#10005;';
          }, 3000);
          return;
        }

        clearTimeout(btn._disarmTimer);
        const p = state.players.find(x => x.id === id);
        if (!p) return;
        state.players = state.players.filter(x => x.id !== id);
        if (state.activeId === id) {
          state.activeId = state.players[0].id;
        }
        lsSave();
        renderRosterList();
      });
    });
  }

  document.getElementById('manage-roster-btn').addEventListener('click', openRosterModal);
  document.getElementById('modal-close-btn').addEventListener('click', closeRosterModal);
  rosterModal.addEventListener('click', e => {
    if (e.target === rosterModal) closeRosterModal();
  });
  document.getElementById('modal-add-btn').addEventListener('click', () => {
    addNewPlayer();
    renderRosterList();
  });

  document.addEventListener('keydown', e => {
    if (e.key === 'Escape' && rosterModal.classList.contains('show')) {
      closeRosterModal();
    }
  });

  document.getElementById('commit-btn').addEventListener('click', () => {
    state.undoStack = state.undoStack || [];
    state.undoStack.push(JSON.parse(JSON.stringify({
      players: state.players, round: state.round,
      gameOver: state.gameOver, activeId: state.activeId
    })));
    if (state.undoStack.length > 10) state.undoStack.shift();

    state.players.forEach(p => {
      const s = computeScore(p);
      p.history.push({
        score: s.total, busted: p.current.busted, flip7: p.current.flip7,
        numbers: [...p.current.numbers], modifiers: [...p.current.modifiers]
      });
      p.total += s.total;
      p.current = emptyCurrent();
    });

    state.round++;
    if (state.players.some(p => p.total >= TARGET)) state.gameOver = true;

    lsSave();
    render();
    window.scrollTo({ top: 0, behavior: 'smooth' });
  });

  document.getElementById('undo-btn').addEventListener('click', () => {
    state.undoStack = state.undoStack || [];
    if (!state.undoStack.length) return;
    const last = state.undoStack.pop();
    state.players = last.players;
    state.round = last.round;
    state.gameOver = false;
    state.activeId = last.activeId;
    lsSave();
    render();
  });

  document.getElementById('reset-btn').addEventListener('click', e => {
    twoTapConfirm(e.currentTarget, '!', () => {
      const names = state.players.map(p => p.name);
      defaultState();
      state.players = names.map((n, i) => ({
        id: i + 1, name: n, total: 0, history: [], current: emptyCurrent()
      }));
      state.activeId = 1;
      lsSave();
      render();
    });
  });

  if (!lsLoad()) defaultState();
  state.players.forEach(p => {
    if (!p.current) p.current = emptyCurrent();
    if (typeof p.total !== 'number') p.total = 0;
    if (!p.history) p.history = [];
  });
  if (!state.undoStack) state.undoStack = [];
  render();
</script>
